DELIMITER //

USE FPLDatabase;

CREATE OR REPLACE PROCEDURE SP_Players_GetById(IN PlayerId INT)

BEGIN

	SELECT 
   	p.ChanceOfPlayingNextRound
   	, p.ChanceOfPlayingThisRound
   	, p.`Code`
   	, p.CostChangeEvent
   	, p.CostChangeEventFall
   	, p.CostChangeStart
   	, p.CostChangeStartFall
   	, p.DreamteamCount
   	, p.ElementType
   	, p.EpNext
   	, p.EpThis
   	, p.EventPoints
	, p.FirstName
   	, p.Form
   	, p.Id
   	, p.InDreamteam
   	, p.News
   	, p.NewsAdded
   	, p.NowCost
   	, p.Photo
   	, p.PointsPerGame
   	, p.SecondName
   	, p.SelectedByPercent
   	, p.Special
   	, p.`Status`
   	, p.Team
   	, p.TeamCode
   	, p.TotalPoints
   	, p.TransfersIn
   	, p.TransfersInEvent
   	, p.TransfersOut
   	, p.TransfersOutEvent
   	, p.ValueForm
   	, p.ValueSeason
   	, p.WebName
   	, p.Minutes
   	, p.GoalsScored
   	, p.Assists
   	, p.CleanSheets
   	, p.GoalsConceded
   	, p.OwnGoals
   	, p.PenaltiesSaved
   	, p.PenaltiesMissed
   	, p.YellowCards
   	, p.RedCards
   	, p.Saves
   	, p.Bonus
   	, p.Bps
   	, p.Influence
   	, p.Creativity
   	, p.IctIndex
   	, p.TeamName
	FROM PLAYERS p
	WHERE p.Id = playerId;

END; //

DELIMITER ;
