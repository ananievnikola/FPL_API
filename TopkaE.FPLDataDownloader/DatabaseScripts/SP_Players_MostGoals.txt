DELIMITER //

USE FPLDatabase;

CREATE OR REPLACE PROCEDURE SP_Players_MostGoals(IN Top INT)

BEGIN

	IF Top IS NULL THEN 
	
	SELECT 
		p.Id
		, p.FirstName
		, p.SecondName
		, p.TeamName
		, p.GoalsScored
		, p.Assists
	FROM PLAYERS p
	ORDER BY p.GoalsScored DESC;
	
	ELSE
	
	SELECT 
		p.Id
		, p.FirstName
		, p.SecondName
		, p.TeamName
		, p.GoalsScored
		, p.Assists
	FROM PLAYERS p
	ORDER BY p.GoalsScored DESC
	LIMIT Top;
	
	END IF; 

END; //

DELIMITER ;
