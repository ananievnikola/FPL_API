DELIMITER //

USE FPLDatabase;

CREATE OR REPLACE PROCEDURE SP_Players_MostTransferedOut(IN Top INT)

BEGIN

	IF Top IS NULL THEN 
	
	SELECT 
		p.Id
		, p.FirstName
		, p.SecondName
		, p.TeamName
		, p.TransfersInEvent
		, p.TransfersOutEvent
		, p.SelectedByPercent
	FROM PLAYERS p
	ORDER BY p.TransfersOutEvent DESC;
	
	ELSE
	
	SELECT 
		p.Id
		, p.FirstName
		, p.SecondName
		, p.TeamName
		, p.TransfersInEvent
		, p.TransfersOutEvent
		, p.SelectedByPercent
	FROM PLAYERS p
	ORDER BY p.TransfersOutEvent DESC
	LIMIT Top;
	
	END IF; 

END; //

DELIMITER ;
