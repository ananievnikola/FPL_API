DELIMITER //

USE FPLDatabase;

CREATE OR REPLACE PROCEDURE SP_Players_MostGoalsInvovement(IN Top INT)

BEGIN

IF Top IS NULL THEN 
	
	SELECT 
		p.Id
		, p.FirstName
		, p.SecondName
		, p.TeamName
		, p.GoalsScored
		, p.Assists
		, CAST(SUM(p.GoalsScored + p.Assists) AS INT) AS Overall
	FROM PLAYERS p
	GROUP BY 
		p.Id
		, p.FirstName
		, p.SecondName
		, p.TeamName
		, p.GoalsScored
		, p.Assists
	ORDER BY Overall DESC, p.GoalsScored DESC, p.Assists DESC;
	
ELSE
	
	SELECT 
		p.Id
		, p.FirstName
		, p.SecondName
		, p.TeamName
		, p.GoalsScored
		, p.Assists
		, CAST(SUM(p.GoalsScored + p.Assists) AS INT) AS Overall
	FROM PLAYERS p
	GROUP BY 
		p.Id
		, p.FirstName
		, p.SecondName
		, p.TeamName
		, p.GoalsScored
		, p.Assists
	ORDER BY Overall DESC, p.GoalsScored DESC, p.Assists DESC
	LIMIT Top;
	
	END IF; 
	

END; //

DELIMITER ;
