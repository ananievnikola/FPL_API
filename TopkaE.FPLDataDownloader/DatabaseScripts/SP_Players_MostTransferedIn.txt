DELIMITER //

CREATE OR REPLACE PROCEDURE SP_Players_MostTransferedIn(IN Top INT)

BEGIN

	IF Top IS NULL THEN 
	
	SELECT 
		e.Id
		, e.FirstName
		, e.SecondName
		, e.TeamName
		, e.TransfersInEvent
		, e.TransfersOutEvent
		, e.SelectedByPercent
	FROM ELEMENTS e
	ORDER BY e.TransfersInEvent DESC;
	
	ELSE
	
	SELECT 
		e.Id
		, e.FirstName
		, e.SecondName
		, e.TeamName
		, e.TransfersInEvent
		, e.TransfersOutEvent
		, e.SelectedByPercent
	FROM ELEMENTS e
	ORDER BY e.TransfersInEvent DESC
	LIMIT Top;
	
	END IF; 

END; //

DELIMITER ;
